Columns = fname, lname, contact, password

CREATE DATABASE lost_found;

USE lost_found;

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fname VARCHAR(50),
    lname VARCHAR(50),
    contact VARCHAR(20),
    password VARCHAR(100)
);

CREATE TABLE items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100),
    description TEXT,
    category VARCHAR(50),
    status VARCHAR(20),
    contact VARCHAR(100)
);
